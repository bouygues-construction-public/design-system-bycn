<Canvas>
  <div
    style={{
      display: 'flex',
      flexWrap: 'wrap',
      justifyContent: 'center',
    }}
  >
    {Object.keys(masoniIconsFilled).map((icon) => (
      <div
        key={icon}
        className="icon-tile"
        onClick={() => {
          const textArea = document.createElement('textarea');
          textArea.value = icon;
          document.body.appendChild(textArea);
          textArea.select();
          document.execCommand('Copy');
          textArea.remove();
        }}
        title={`Copy ${icon} to clipboard`}
      >
        <span
          aria-label={icon}
          className={`mas-${icon}`}
          style={{
            fontSize: 32,
            display: 'flex',
            justifyContent: 'center',
            alignItems: 'flex-end',
            height: '100%',
            marginBottom: 8,
          }}
        ></span>
        <span
          style={{
            fontSize: 15,
            display: 'flex',
            justifyContent: 'center',
            alignItems: 'flex-start',
            height: '100%',
          }}
        >
          {icon}
        </span>
      </div>
    ))}
  </div>
</Canvas>
